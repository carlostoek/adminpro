## Critical Issues to Fix

### C-007: Atomicity Failure en Cambios de Rol (user_management.py)
**Problem:** If Telegram API fails after DB update, user has new role in DB but old role in channel.
**Current flow:**
1. UPDATE User.role in DB
2. Call Telegram API (ban/unban)
3. If step 2 fails, DB is already committed

**Fix:** Telegram first, then DB - or use compensating transactions

### C-008: Transacción Larga con Llamadas Telegram (subscription.py)
**Problem:** Long transactions hold locks while waiting for Telegram API responses.
**Current flow:**
1. BEGIN TRANSACTION
2. SELECT/UPDATE records
3. Call Telegram API (slow, seconds)
4. COMMIT

**Fix:** Separate into phases: SELECT → API → UPDATE
</context>

```python
async def change_user_role(
    self,
    user_id: int,
    new_role: UserRole,
    changed_by: int
) -> Tuple[bool, str]:
    """
    Cambia el rol de un usuario con atomicidad garantizada.

    Pattern: Telegram first, then DB. If Telegram fails, DB never changes.
    If DB fails after Telegram, we have a log but user can be fixed manually.

    Args:
        user_id: ID del usuario a modificar
        new_role: Nuevo rol
        changed_by: ID del admin que realiza el cambio

    Returns:
        Tuple of (success, message)
    """
    try:
        # Step 0: Validation (read-only, no DB changes yet)
        can_modify, error_msg = await self._can_modify_user(
            target_user_id=user_id,
            admin_user_id=changed_by
        )
        if not can_modify:
            return (False, error_msg)

        # Get current state (read-only)
        stmt = select(User).where(User.user_id == user_id)
        result = await self.session.execute(stmt)
        user = result.scalar_one_or_none()

        if not user:
            return (False, "Usuario no encontrado")

        old_role = user.role
        if old_role == new_role:
            return (True, f"El usuario ya tiene el rol {new_role.value}")

        # Step 1: Determine channel operations needed
        from bot.services.channel import ChannelService
        channel_service = ChannelService(self.session, self.bot)

        vip_channel_id = await channel_service.get_vip_channel_id()
        free_channel_id = await channel_service.get_free_channel_id()

        # Step 2: Execute Telegram operations FIRST (before any DB writes)
        # If this fails, no DB changes have been made
        telegram_ops_success = True
        telegram_errors = []

        try:
            if new_role == UserRole.FREE and old_role == UserRole.VIP:
                # VIP -> FREE: Remove from VIP channel
                if vip_channel_id:
                    await self.bot.ban_chat_member(
                        chat_id=vip_channel_id,
                        user_id=user_id
                    )
                    logger.info(f"User {user_id} banned from VIP channel (role change)")

            elif new_role == UserRole.VIP and old_role == UserRole.FREE:
                # FREE -> VIP: This usually happens via token redemption, not direct change
                # But if direct, user will need invite link
                pass

            elif new_role == UserRole.ADMIN:
                # Promoting to admin - no channel changes needed
                pass

            # Add other role transition logic as needed

        except Exception as e:
            telegram_ops_success = False
            telegram_errors.append(str(e))
            logger.error(f"Telegram operations failed for role change: {e}")

        if not telegram_ops_success:
            # Telegram failed - don't proceed with DB update
            # This keeps DB and Telegram in sync (old role in both)
            return (
                False,
                f"Error al actualizar canales de Telegram: {'; '.join(telegram_errors)}. "
                "El rol no ha sido cambiado. Inténtalo de nuevo."
            )

        # Step 3: Now safe to update DB (Telegram operations succeeded)
        user.role = new_role
        user.updated_at = datetime.utcnow()

        # Log the change
        role_change_service = RoleChangeService(self.session)
        await role_change_service.log_role_change(
            user_id=user_id,
            new_role=new_role,
            changed_by=changed_by,
            reason=RoleChangeReason.MANUAL_CHANGE,
            previous_role=old_role,
            change_source="ADMIN_PANEL"
        )

        # Flush to ensure DB is updated
        await self.session.flush()

        logger.info(
            f"Role changed: user {user_id} {old_role.value} -> {new_role.value} "
            f"by admin {changed_by}"
        )

        return (True, f"Rol cambiado de {old_role.value} a {new_role.value}")

    except Exception as e:
        # Unexpected error - log and return error
        logger.error(f"Error in change_user_role: {e}", exc_info=True)
        return (False, f"Error al cambiar rol: {str(e)}")
```

2. If Telegram fails, return error without touching DB
3. Only update DB after Telegram succeeds
4. Clear error messages for each failure mode

```python
async def expel_user_from_channels(
    self,
    user_id: int,
    expelled_by: int
) -> Tuple[bool, str, List[str]]:
    """
    Expulsa a un usuario de los canales VIP y Free.

    Pattern: Telegram operations first, then update DB if needed.

    Args:
        user_id: ID del usuario a expulsar
        expelled_by: ID del admin que expulsa

    Returns:
        Tuple of (success, message, expelled_from)
    """
    # Validation (read-only)
    can_modify, error_msg = await self._can_modify_user(
        target_user_id=user_id,
        admin_user_id=expelled_by
    )
    if not can_modify:
        return (False, error_msg, [])

    from bot.services.channel import ChannelService
    channel_service = ChannelService(self.session, self.bot)

    # Get channel IDs
    vip_channel_id = await channel_service.get_vip_channel_id()
    free_channel_id = await channel_service.get_free_channel_id()

    if not vip_channel_id and not free_channel_id:
        return (False, "No hay canales configurados para expulsar", [])

    # Execute Telegram operations
    expelled_from = []
    errors = []

    channels_to_try = []
    if vip_channel_id:
        channels_to_try.append(("VIP", vip_channel_id))
    if free_channel_id:
        channels_to_try.append(("Free", free_channel_id))

    for channel_name, channel_id in channels_to_try:
        try:
            await self.bot.ban_chat_member(
                chat_id=channel_id,
                user_id=user_id
            )
            expelled_from.append(channel_name)
            logger.info(f"User {user_id} expelled from {channel_name} by admin {expelled_by}")
        except Exception as e:
            error_str = str(e).lower()
            # User not in channel is OK
            if "user not found" in error_str or "not a member" in error_str:
                expelled_from.append(channel_name)
                logger.info(f"User {user_id} already not in {channel_name}")
            else:
                errors.append(f"{channel_name}: {str(e)}")
                logger.warning(f"Could not expel user {user_id} from {channel_name}: {e}")

    # Update VIPSubscriber.kicked_from_channel_at if applicable
    if "VIP" in expelled_from:
        try:
            await self.session.execute(
                update(VIPSubscriber)
                .where(VIPSubscriber.user_id == user_id)
                .values(kicked_from_channel_at=datetime.utcnow())
            )
        except Exception as e:
            logger.warning(f"Could not update kicked_from_channel_at for {user_id}: {e}")

    if errors and not expelled_from:
        return (False, f"No se pudo expulsar: {'; '.join(errors)}", [])

    if errors:
        return (
            True,
            f"Expulsado parcialmente de: {', '.join(expelled_from)}. "
            f"Errores: {'; '.join(errors)}",
            expelled_from
        )

    channels_str = ", ".join(expelled_from)
    return (True, f"Usuario expulsado de: {channels_str}", expelled_from)
```

```python
async def approve_ready_free_requests(
    self,
    wait_time_minutes: int,
    free_channel_id: str
) -> Tuple[int, int]:
    """
    Process free channel requests with separated transaction phases.

    Phase 1: Short transaction - claim requests atomically
    Phase 2: API calls - Telegram operations (no DB locks held)
    Phase 3: Short transaction - update results

    This prevents holding DB locks during slow Telegram API calls.
    """
    cutoff_time = datetime.utcnow() - timedelta(minutes=wait_time_minutes)

    # PHASE 1: Short transaction to claim requests
    claimed_request_ids = []

    async with self.session.begin():
        result = await self.session.execute(
            select(FreeChannelRequest)
            .where(
                FreeChannelRequest.processed == False,
                FreeChannelRequest.request_date <= cutoff_time
            )
            .order_by(FreeChannelRequest.request_date)
            .limit(100)
            .with_for_update(skip_locked=True)
        )
        requests_to_process = result.scalars().all()

        # Mark as processed immediately (short transaction)
        for request in requests_to_process:
            request.processed = True
            request.processed_at = datetime.utcnow()
            claimed_request_ids.append(request.user_id)

    if not claimed_request_ids:
        return (0, 0)

    logger.info(f"Claimed {len(claimed_request_ids)} requests for processing")

    # PHASE 2: API calls (no DB transaction active)
    # This can take seconds but doesn't hold any DB locks
    success_count = 0
    error_count = 0
    failed_user_ids = []

    for user_id in claimed_request_ids:
        try:
            await self.bot.approve_chat_join_request(
                chat_id=free_channel_id,
                user_id=user_id
            )
            success_count += 1
            logger.info(f"Approved free request for user {user_id}")

        except Exception as e:
            error_count += 1
            failed_user_ids.append(user_id)
            logger.error(f"Failed to approve request for {user_id}: {e}")

    # PHASE 3: Log failures (optional, short transaction)
    if failed_user_ids:
        logger.warning(f"Failed to approve {len(failed_user_ids)} requests: {failed_user_ids}")

    return (success_count, error_count)
```

Key improvement: Use explicit transaction blocks to ensure locks are released before API calls.

```python
async def kick_expired_vip_from_channel(self, channel_id: str) -> Tuple[int, int, int]:
    """
    Kick expired VIPs with separated transaction phases.

    Phase 1: Short transaction - get and mark VIPs as being processed
    Phase 2: API calls - Telegram ban operations
    Phase 3: Short transaction - update kicked status
    """
    # PHASE 1: Short transaction to get VIPs to kick
    vips_to_kick = []  # List of (user_id, vip_id) tuples

    async with self.session.begin():
        result = await self.session.execute(
            select(VIPSubscriber)
            .where(
                VIPSubscriber.status == "expired",
                VIPSubscriber.kicked_from_channel_at.is_(None)
            )
            .order_by(VIPSubscriber.expiry_date)
            .limit(100)
        )
        expired_vips = result.scalars().all()

        for vip in expired_vips:
            vips_to_kick.append((vip.user_id, vip.id))

    if not vips_to_kick:
        return (0, 0, 0)

    # PHASE 2: API calls (no DB transaction)
    kicked_user_ids = []
    failed_user_ids = []
    already_out_user_ids = []

    for user_id, vip_id in vips_to_kick:
        try:
            await self.bot.ban_chat_member(
                chat_id=channel_id,
                user_id=user_id
            )
            kicked_user_ids.append((user_id, vip_id))
            logger.info(f"Banned expired VIP {user_id} from channel")

        except Exception as e:
            error_str = str(e).lower()
            if "user not found" in error_str or "not a member" in error_str:
                already_out_user_ids.append((user_id, vip_id))
                logger.info(f"User {user_id} already not in channel")
            else:
                failed_user_ids.append(user_id)
                logger.warning(f"Could not ban {user_id}: {e}")

    # PHASE 3: Short transaction to update kicked status
    kicked_count = 0
    already_kicked_count = 0

    async with self.session.begin():
        # Mark successfully kicked users
        for user_id, vip_id in kicked_user_ids + already_out_user_ids:
            await self.session.execute(
                update(VIPSubscriber)
                .where(VIPSubscriber.id == vip_id)
                .values(kicked_from_channel_at=datetime.utcnow())
            )

        kicked_count = len(kicked_user_ids)
        already_kicked_count = len(already_out_user_ids)

    failed_count = len(failed_user_ids)

    return (kicked_count, already_kicked_count, failed_count)
```

This ensures DB locks are never held during Telegram API calls.
