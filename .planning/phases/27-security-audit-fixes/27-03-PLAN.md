### C-005: TOCTOU en advance_stage VIP (lines ~87-143)
**Problem:** Stage progression can happen multiple times if concurrent requests.
**Current flow:**
1. SELECT subscriber and check current stage
2. (Time window for race)
3. UPDATE stage

**Fix:** UPDATE condicional con rowcount check

**Problem:** Token can be validated multiple times during the flow.
**Current flow:**
1. SELECT check token exists and stage == 3
2. (Time window for race)
3. Allow access

**Fix:** Marcar token "en uso" inmediatamente con UPDATE

```python
async def advance_stage(self, user_id: int, from_stage: int) -> bool:
    # Get subscriber
    result = await self.session.execute(
        select(VIPSubscriber).where(VIPSubscriber.user_id == user_id)

    subscriber = result.scalar_one_or_none()

    # Validate subscription not expired (TOCTOU window here)

    current_stage = subscriber.vip_entry_stage if subscriber.vip_entry_stage else 0

    next_stage = from_stage + 1
```

Replace with atomic pattern:
```python
from sqlalchemy import update, select

async def advance_stage(self, user_id: int, from_stage: int) -> Tuple[bool, str]:
    """
    Advance VIP entry stage with race condition protection.

    Uses atomic UPDATE with conditional WHERE clause.

    Args:
        user_id: ID del usuario
        from_stage: Etapa actual esperada (para validación)

    Returns:
        Tuple of (success, message)
    """
    # Validate stage progression logic first
    if from_stage not in (1, 2):
        return (False, f"No se puede avanzar desde la etapa {from_stage}")

    next_stage = from_stage + 1

    # Atomic UPDATE: only update if conditions match
    # This prevents race conditions where another request already advanced the stage
    result = await self.session.execute(
        update(VIPSubscriber)
        .where(
            VIPSubscriber.user_id == user_id,
            VIPSubscriber.vip_entry_stage == from_stage,  # Must match expected stage
            VIPSubscriber.expiry_date > datetime.utcnow(),  # Subscription not expired
            VIPSubscriber.status == "active"  # Subscription active
        )
        .values(vip_entry_stage=next_stage)
    )

    if result.rowcount == 0:
        # UPDATE failed - check why for better error message
        subscriber_result = await self.session.execute(
            select(VIPSubscriber).where(VIPSubscriber.user_id == user_id)
        )
        subscriber = subscriber_result.scalar_one_or_none()

        if not subscriber:
            return (False, "Suscripción VIP no encontrada")

        if subscriber.is_expired():
            return (False, "Tu suscripción VIP ha expirado")

        current = subscriber.vip_entry_stage or 0
        if current != from_stage:
            if current > from_stage:
                return (False, f"Ya has avanzado a la etapa {current}")
            else:
                return (False, f"Etapa incorrecta. Actual: {current}, Esperada: {from_stage}")

        return (False, "No se pudo avanzar la etapa")

    logger.info(f"User {user_id} VIP entry advanced: stage {from_stage} -> {next_stage}")
    return (True, f"Etapa {next_stage} desbloqueada")
```

Key changes:
1. Use UPDATE with WHERE clause instead of SELECT then UPDATE
2. Check rowcount to verify the UPDATE succeeded
3. Include all validation in WHERE clause (stage, expiry, status)
4. Return detailed error messages for debugging
5. Change return type from bool to Tuple[bool, str]

```python
from sqlalchemy import update

async def validate_and_consume_entry_token(self, token: str) -> Tuple[bool, str, Optional[int]]:
    """
    Validate and atomically consume an entry token.

    This prevents race conditions where the same token is used multiple times.

    Args:
        token: Token de entrada a validar y consumir

    Returns:
        Tuple of (success, message, user_id):
        - success: True si el token es válido y fue consumido
        - message: Mensaje de éxito o error
        - user_id: ID del usuario si exitoso, None si falló
    """
    # Atomic UPDATE: clear token and stage only if token matches and stage is 3
    # This ensures the token can only be used once
    result = await self.session.execute(
        update(VIPSubscriber)
        .where(
            VIPSubscriber.vip_entry_token == token,
            VIPSubscriber.vip_entry_stage == 3,
            VIPSubscriber.expiry_date > datetime.utcnow()
        )
        .values(
            vip_entry_stage=None,  # Flow complete
            vip_entry_token=None   # Token consumed
        )
    )

    if result.rowcount == 0:
        # Check why it failed for better error message
        subscriber_result = await self.session.execute(
            select(VIPSubscriber).where(VIPSubscriber.vip_entry_token == token)
        )
        subscriber = subscriber_result.scalar_one_or_none()

        if not subscriber:
            return (False, "Token de entrada inválido", None)

        if subscriber.is_expired():
            return (False, "Tu suscripción VIP ha expirado", None)

        if subscriber.vip_entry_stage != 3:
            if subscriber.vip_entry_stage is None:
                return (False, "El acceso ya fue completado anteriormente", None)
            return (False, f"Completa la etapa {subscriber.vip_entry_stage} primero", None)

        return (False, "Token no válido", None)

    # Get user_id for the success case
    subscriber_result = await self.session.execute(
        select(VIPSubscriber.user_id).where(VIPSubscriber.vip_entry_token.is_(None))
        .order_by(VIPSubscriber.updated_at.desc())
        .limit(1)
    )
    user_id = subscriber_result.scalar_one()

    logger.info(f"Entry token consumed for user {user_id}")
    return (True, "Acceso VIP confirmado", user_id)


async def is_entry_token_valid(self, token: str) -> bool:
    """
    Check if token is valid (non-destructive check for UI display).

    Note: This does NOT consume the token. Use validate_and_consume_entry_token
    for actual access validation.
    """
    result = await self.session.execute(
        select(VIPSubscriber).where(
            VIPSubscriber.vip_entry_token == token,
            VIPSubscriber.vip_entry_stage == 3,
            VIPSubscriber.expiry_date > datetime.utcnow()
        )
    )
    subscriber = result.scalar_one_or_none()
    return subscriber is not None
```

Key changes:
1. Add new validate_and_consume_entry_token method for atomic consumption
2. UPDATE clears both stage and token atomically
3. Keep is_entry_token_valid for non-destructive UI checks
4. Clear documentation on which method to use when
