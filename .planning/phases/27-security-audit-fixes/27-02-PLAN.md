## Critical Issues to Fix

### C-003: Inconsistencia DB/Telegram en kick_expired_vip_from_channel (lines 557-601)
**Problem:** Users expired but not banned become orphaned - in DB as expired but still in channel.
**Current flow:**
1. Query expired VIPs
2. Try to kick from Telegram
3. If kick fails, user stays in channel but DB shows expired

**Fix:** Add kicked_from_channel tracking field + retry logic

### C-004: Race Condition en approve_ready_free_requests (lines 1005-1156)
**Problem:** Same request can be processed by multiple concurrent workers.
**Current flow:**
1. SELECT pending requests
2. (Time window for race)
3. Mark as processed
4. Call Telegram API

**Fix:** UPDATE with RETURNING to atomically claim requests
</context>

from sqlalchemy import update, select
from sqlalchemy.dialects.postgresql import returning

async def approve_ready_free_requests(
    self,
    wait_time_minutes: int,
    free_channel_id: str
) -> Tuple[int, int]:
    """
    Process free channel requests with race condition protection.

    Uses UPDATE with RETURNING to atomically claim requests.

    Returns:
        Tuple of (success_count, error_count)
    """
    cutoff_time = datetime.utcnow() - timedelta(minutes=wait_time_minutes)

    # Method: Use SKIP LOCKED to get requests not being processed by others
    # For PostgreSQL: UPDATE ... RETURNING with SKIP LOCKED
    # For SQLite: Use processed flag with immediate update

    # Step 1: Atomically mark requests as "being processed" and get their IDs
    # This prevents other workers from picking them up

    if self._is_postgresql():
        # PostgreSQL: UPDATE with RETURNING
        update_stmt = (
            update(FreeChannelRequest)
            .where(
                FreeChannelRequest.processed == False,
                FreeChannelRequest.request_date <= cutoff_time
            )
            .values(processed=True, processed_at=datetime.utcnow())
            .returning(FreeChannelRequest.user_id)
        )

        # Add SKIP LOCKED to skip rows locked by other transactions
        # Note: SQLAlchemy doesn't directly support SKIP LOCKED in UPDATE
        # Use raw SQL or separate SELECT FOR UPDATE SKIP LOCKED + UPDATE

    # Cross-database approach: Use SELECT FOR UPDATE SKIP LOCKED then UPDATE

    # Get pending requests with row locking (skip locked rows)
    result = await self.session.execute(
        select(FreeChannelRequest)
        .where(
            FreeChannelRequest.processed == False,
            FreeChannelRequest.request_date <= cutoff_time
        )
        .order_by(FreeChannelRequest.request_date)
        .limit(100)  # Process in batches
        .with_for_update(skip_locked=True)  # Skip rows locked by other workers
    )
    requests_to_process = result.scalars().all()

    if not requests_to_process:
        return (0, 0)

    # Mark these as processed immediately (still in transaction)
    for request in requests_to_process:
        request.processed = True
        request.processed_at = datetime.utcnow()

    await self.session.flush()  # Ensure updates are visible

    # Now process each request (outside the critical section)
    success_count = 0
    error_count = 0

    for request in requests_to_process:
        try:
            await self.bot.approve_chat_join_request(
                chat_id=free_channel_id,
                user_id=request.user_id
            )
            success_count += 1
            logger.info(f"Approved free request for user {request.user_id}")

        except Exception as e:
            error_count += 1
            logger.error(f"Failed to approve request for {request.user_id}: {e}")

            # Don't rollback - request is marked as processed to prevent retry
            # Log the error for manual investigation

    return (success_count, error_count)
```

Key changes:
1. Use SELECT FOR UPDATE SKIP LOCKED to claim requests atomically
2. Mark as processed BEFORE calling Telegram API
3. Process in batches with LIMIT
4. Don't rollback on Telegram errors - prevents infinite retry loops

async def kick_expired_vip_from_channel(self, channel_id: str) -> Tuple[int, int, int]:
    """
    Kick expired VIPs from channel with tracking.

    Returns:
        Tuple of (kicked_count, already_kicked_count, failed_count)
    """
    from sqlalchemy import update

    # Query expired VIPs not yet kicked (using new tracking field)
    result = await self.session.execute(
        select(VIPSubscriber)
        .where(
            VIPSubscriber.status == "expired",
            VIPSubscriber.kicked_from_channel_at.is_(None)  # Not yet kicked
        )
        .order_by(VIPSubscriber.expiry_date)
        .limit(100)  # Batch processing
    )
    expired_vips = result.scalars().all()

    if not expired_vips:
        return (0, 0, 0)

    kicked_count = 0
    failed_count = 0

    for vip in expired_vips:
        try:
            # Attempt to kick from Telegram
            await self.bot.ban_chat_member(
                chat_id=channel_id,
                user_id=vip.user_id
            )

            # Mark as successfully kicked
            vip.kicked_from_channel_at = datetime.utcnow()
            kicked_count += 1

            logger.info(f"Kicked expired VIP {vip.user_id} from channel")

        except Exception as e:
            # Check if user is already not in channel (which is fine)
            error_str = str(e).lower()
            if "user not found" in error_str or "user is not a member" in error_str:
                # User already not in channel - mark as done
                vip.kicked_from_channel_at = datetime.utcnow()
                logger.info(f"User {vip.user_id} already not in channel")
            else:
                # Real error - will retry next run
                failed_count += 1
                logger.warning(f"Could not kick {vip.user_id}: {e}")

    await self.session.flush()

    # Count already kicked for reporting
    already_kicked_result = await self.session.execute(
        select(func.count(VIPSubscriber.id))
        .where(
            VIPSubscriber.status == "expired",
            VIPSubscriber.kicked_from_channel_at.isnot(None)
        )
    )
    total_kicked = already_kicked_result.scalar_one()

    return (kicked_count, total_kicked - kicked_count, failed_count)
```

Key changes:
1. Use kicked_from_channel_at timestamp to track progress
2. Process in batches with LIMIT
3. Handle "already not in channel" as success
4. Return detailed counts for monitoring
5. Failed kicks remain in queue for next run
